Quick Wins (1 day each)

1. Browser Testing Setup

# Install playwright for multiple browsers
npm install -D @playwright/test

# Update playwright.config.ts
export default {
  projects: [
    { name: 'chromium', use: { ...devices['Desktop Chrome'] } },
    { name: 'firefox', use: { ...devices['Desktop Firefox'] } },
    { name: 'webkit', use: { ...devices['Desktop Safari'] } },
  ]
};

# Run tests
npx playwright test --project=firefox
npx playwright test --project=webkit

2. TypeScript Definitions

// pan.d.ts (add to package root)
export * from './src/core/pan-bus.mjs';
export * from './src/core/pan-client.mjs';

// package.json
{
  "types": "./pan.d.ts",
  "exports": {
    ".": {
      "types": "./pan.d.ts",
      "default": "./src/pan.mjs"
    }
  }
}

3. npm Publication

# Create packages/bus/package.json
{
  "name": "@pan/bus",
  "version": "1.0.0",
  "type": "module",
  "main": "./dist/pan-bus.mjs",
  "types": "./dist/pan-bus.d.ts",
  "exports": {
    ".": {
      "types": "./dist/pan-bus.d.ts",
      "default": "./dist/pan-bus.mjs"
    }
  }
}

# Publish
npm login
